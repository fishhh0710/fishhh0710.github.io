CSES Longest Palindrome 解題心得

2023/11/20

解法

據我所知，有兩個比較普遍的作法

  1. Manacher Algorithm（馬拉車演算法 -ChatGPT 給我的）
  2. Rolling Hash（我的做法）

解題過程

如果要看教學文可以去找大神寫的 這篇 我覺得寫得超好

有了這一篇文的基礎後，相信你已經會做 Rolling Hash 了！

那，這一題要如何用 Rolling Hash 做呢？
可以知道一件事情，一個迴文代表兩邊是對稱的，如果是一個迴文就會滿足從左到右看以及從右到左看是一樣的。
那麼兩邊看過來的 Hash 值就會相同。所以我們需要分別對兩個方向各做一次 Rolling Hash。

接下來就是枚舉中點，然後對它做二分搜。因為這個具有單調性（如果延伸 不行，那麼延伸比 長保證就不行）

我的 code（只有枚舉奇數長度的部分）

int fastpow(int a,int b){
    int ret = 1;
    while(b){
        if(b%2){
            ret = (a*ret)%P;
        }
        a = (a*a)%P;
        b/=2;
    }
    return ret%P;
}
int hash(int l,int r){
    int d = ((h[r]-h[l-1])%P+P)%P;
    return (d*fastpow(fastpow(x1,l),P-2))%P;

int hash2(int l,int r){
    int d = ((h2[r]-h2[l-1])%P+P)%P;
    return (d*fastpow(fastpow(x1,l),P-2))%P;
}

rep1(i,n){ //枚舉中點

評估一下複雜度， ，高達 會 TLE。

優化

我決定從 Hash 的比較 function 下手。可以發現，其中的 fastpow(x1,l) 一直都是一樣的，所以可以對它建表。
這樣只降了一個 log 還是很危。
接下來，因為 fastpow(fastpow(x1,l),P-2) 只要 一樣那麼出來的值也會一樣。所以，我們就開一個長度為 的陣列紀錄對於每一個 的 fastpow(fastpow(x1,l),P-2) 值就好了！！！

我完整的 code

const int P = 1e9+7;
int h[2000100],h2[2000100],tbbb[2000100];
int x1 = 43;
int fastpow(int a,int b){
    int ret = 1;
    while(b){
        if(b%2){
            ret = (a*ret)%P;
        }
        a = (a*a)%P;
        b/=2;
    }
    return ret%P;
}
int chsh(int l,int r){
    int d = ((h[r]-h[l-1])%P+P)%P;
    return (d*tbbb[l])%P;
}
int chsh2(int l,int r){
    int d = ((h2[r]-h2[l-1])%P+P)%P;
    return (d*tbbb[l])%P;
}
void solve(){
    int n;
    string s;
    cin>>s;
    n = s.size();
    int x = x1;
    // build table
    ptab[1] = x1;
    tbbb[1] = fastpow(ptab[1],P-2);
    for(int i=2;i<=n;i++){
        ptab[i] = (ptab[i-1]*x1)%P;
        tbbb[i] = fastpow(ptab[i],P-2);
    }
    // rolling hash
    for(int i=1;i<=n;i++){
        h[i] = (s[i-1]*x)%P;
        x = (x*x1)%P;
    }
    rep1(i,n){
        h[i] += h[i-1];
        h[i]%=P;
    }
    reverse(all(s));
    x = x1;
    for(int i=1;i<=n;i++){
        h2[i] = (s[i-1]*x)%P;
        x = (x*x1)%P;
    }
    rep1(i,n){
        h2[i] += h2[i-1];
        h2[i]%=P;
    }

    int ans = 0,al = 1,ar = 1;
    rep1(i,n){
        int maxd = min(i-1,n-i);
        if(2*maxd+1<=ans)continue;
        int add = (1<<25),now = 0;
        while(add){
            int nxt = now+add;
            if(nxt>maxd){
                add/=2;
                continue;
            }
            int ha = chsh(i-nxt,i); //i-nxt,i
            int hb = chsh2(n-i-nxt+1,n-i+1); //i,i+nxt
            if(ha==hb){
                now+=add;
            }
            else{
                add/=2;
            }
        }
        if(2*now+1>ans){
            ans = 2*now+1;
            al = i-now;
            ar = i+now;
        }
    }
 
    rep1(i,n){
        int add = (1<<25);
        int now = 0;
        int maxd = min(i-1,n-i+1);
        if(2*maxd<=ans)continue;
        while(add){
            int nxt = now+add;
            if(nxt>maxd){
                add/=2;
                continue;
            }
            int ha = chsh(i-nxt,i-1); //i-nxt,i-1
            int hb = chsh2(n-i-nxt+2,n-i+1); //i,i+nxt-1
            if(ha==hb){
                now+=add;
            }
            else{
                add/=2;
            }
        }
        if(2*(now)>ans){
            ans = 2*now;
            al = i-now;
            ar = i+now-1;
        }
    }
 
    reverse(all(s));
    for(int i=al;i<=ar;i++){
        cout<<s[i-1];
    }
}
signed main(){
    AC
    solve();
}
